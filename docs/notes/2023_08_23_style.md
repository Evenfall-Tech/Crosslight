## C/C++ Code Style

### CMake

```cmake
add_library(
    ${BINARY_NAME} MODULE
    src/main.cpp
    ${PUBLIC_HEADERS}
    ${PRIVATE_HEADERS}
)

target_link_libraries(${BINARY_NAME} PRIVATE mylib myotherlib)
target_include_directories(${BINARY_NAME} PRIVATE
    ${SOME_DIR}
)
set_target_properties(
    ${BINARY_NAME} PROPERTIES
	  SOME_PROPERTY "property"
)
```
